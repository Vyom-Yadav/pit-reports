33

34

35 
 * </p>
36 
 * <pre>
37 
 * return (x);          // parens around identifier
38

39

40

41

42

43

44 
 * boolean b = (~a) &gt; -27            // parens around ~a
45

46 
 * </pre>
47 
 * <p>
48 
 * The check is not "type aware", that is to say, it can't tell if parentheses
49

50

51

52 
 * </p>
53 
 * <pre>
54 
 * int x = (a + b) + c; // 1st Case
55

56 
 * int q = 4;
57 
 * int r = 3;
58

59

60

61

62

63

64

65

66

67
